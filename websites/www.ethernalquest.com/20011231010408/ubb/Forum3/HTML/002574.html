EQDecode::EQDecode (QObject *parent = 0, const char *name = 0)
: QObject (parent, name)
{
// Link the packet class resetDecoder signal to our ResetDecoder slot
connect (parent, SIGNAL(resetDecoder()), this, SLOT(ResetDecoder()));

// Link our BackfillPlayer signal to the packet class backfillPlayer slot
connect (this, SIGNAL(BackfillPlayer(playerProfileStruct *)),
parent, SLOT(pkt_backfillPlayer(playerProfileStruct *)));

// Link our BackfillSpawn signal to the packet class backfillSpawn slot
connect (this, SIGNAL(BackfillSpawn(spawnStruct *)),
parent, SLOT(pkt_backfillSpawn(spawnStruct *)));

// Link our refreshMap signal to the packet class refreshMap slot
connect (this, SIGNAL(refreshMap()), parent, SLOT(pkt_refreshMap()));

// Our key is unknown
decodeKey = 0;

// We're not locating keys
locateActive = 0;

// Create our queue mutex
pthread_mutex_init (&mutexQueue, NULL);
}

void
EQDecode::ResetDecoder (void)
{
EQPktRec *pkt;
int i;

// Kill our search thread if active
if (locateActive)
{
// Request cancellation and wait for it if successful
if (pthread_cancel(locateThread) == 0)
pthread_join(locateThread, NULL);

// Unlock our mutex (in case thread left it locked)
pthread_mutex_unlock(&mutexQueue);
}

// NOTE: Since we know the locate thread is not running and since we are
// the main thread, we don't need to use the mutex to clear the queues.

// Clear our player packet queue
for (i = 0; i < queuePlayerProfile.size(); i++)
{
pkt = (EQPktRec *)queuePlayerProfile[i];
free(pkt);
}

// Empty the player packet queue
queuePlayerProfile.clear();

// Clear our zone spawns packet queue
for (i = 0; i < queueZoneSpawns.size(); i++)
{
pkt = (EQPktRec *)queueZoneSpawns[i];
free(pkt);
}

// Empty the queue
queueZoneSpawns.clear();

// Clear our spawns packet queue
for (i = 0; i < queueSpawns.size(); i++)
{
pkt = (EQPktRec *)queueSpawns[i];
free(pkt);
}

// Empty the queue
queueSpawns.clear();

// Set our key to zero
decodeKey = 0;

// Let anyone interested know we changed it
emit KeyChanged();

// We're no longer looking
locateActive = 0;
}

void
EQDecode::LocateKey ()
{
unsigned int i;
EQPktRec *player, *spawn;

// Grab our player packet
player = queuePlayerProfile[0];

#if HAVE_LIBEQ
// Find a shiny new key
i = 0;
while (!decodeKey && (i < queueSpawns.size()))
{
spawn = queueSpawns[i];
decodeKey = FindKey(player->data, player->len, spawn->data, spawn->len);
i++;
}
#else
decodeKey = 0;
#endif

// Bail if we didn't find a key at all
if (!decodeKey)
{
locateActive = 0;
return;
}
FoundKey();
}

void
EQDecode::FoundKey ()
{
unsigned int i, j;
EQPktRec *pkt;
struct playerProfileStruct *pdata;
struct zoneSpawnsStruct *zdata;
struct newSpawnStruct *ndata;
struct spawnStruct *sdata;

#if !HAVE_LIBEQ
printf("BUG: libEQ not present, returning from FoundKey()\n");
return;
#endif

// Pass on the good news!
printf("FOUND KEY: 0x%08lx\n", decodeKey);
emit KeyChanged();

// Check to see if we've been cancelled
pthread_testcancel();

// Lock our mutex
pthread_mutex_lock(&mutexQueue);

// Decode our player packet queue
for (i = 0; i < queuePlayerProfile.size(); i++)
{
pkt = (EQPktRec *)queuePlayerProfile[i];
#if HAVE_LIBEQ
if (ProcessPacket(pkt->data, &pkt->len, &decodeKey, ""))
{
pdata = (struct playerProfileStruct *)(pkt->data);
emit BackfillPlayer(pdata);
}
else
#endif
printf("Warning: Failed to decrypt queued Player Profile packet: %d.\n", i);
free(pkt);
}

// Empty the player packet queue
queuePlayerProfile.clear();

// Decode our zone spawns packet queue
for (i = 0; i < queueZoneSpawns.size(); i++)
{
pkt = (EQPktRec *)queueZoneSpawns[i];
#if HAVE_LIBEQ
if (ProcessPacket(pkt->data, &pkt->len, &decodeKey, ""))
{
zdata = (struct zoneSpawnsStruct *)(pkt->data);
for (j = 0; j < (pkt->len - 2) / sizeof(struct spawnStruct); j++)
{
sdata = (struct spawnStruct *)(&zdata->spawn[j]);
emit BackfillSpawn(sdata);
}
}
else
#endif
printf("Warning: Failed to decrypt queued Zone Spawns packet: %d.\n", i);
free(pkt);
}

// Empty the queue
queueZoneSpawns.clear();

// Decode our spawns packet queue
#if HAVE_LIBEQ
for (i = 0; i < queueSpawns.size(); i++)
{
pkt = (EQPktRec *)queueSpawns[i];
DecodeSpawn(pkt->data, pkt->len, decodeKey);
ndata = (struct newSpawnStruct *)(pkt->data);
sdata = (struct spawnStruct *)(&ndata->spawn);
emit BackfillSpawn(sdata);
free(pkt);
}
#endif

// Refresh the map
emit refreshMap ();

// Empty the queue
queueSpawns.clear();

// Unlock our mutex
pthread_mutex_unlock(&mutexQueue);

// Let them know we're done
locateActive = 0;

// Get outa here
return;
}

int
EQDecode: ecodePacket (unsigned char *data, int *len, const char *cli)
{
unsigned int opcode;
EQPktRec *player, *pktrec;
int result;
UDWORD prevKey;

opcode = data[1] | (data[0] << 8);

if (opcode == 0x3620)
data[0] = 0x2d;
else if(opcode == 0x2d20)
data[0] = 0x35;
else if(opcode == 0x2a20)
data[0] = 0x2b;
else if(opcode == 0x2920)
data[0] = 0x2a;
else if(opcode == 0x2c20)
data[0] = 0x36;
else if(opcode == 0xa120)
data[0] = 0x9f;
else if(opcode == 0x9f20)
data[0] = 0xa1;

opcode = data[1] | (data[0] << 8);

// Get the opcode of the current packet
// Check to see if it is a compressed packet
if( ( opcode == CompressedPlayerItemCode ) ||
( opcode == CompressedDoorSpawnCode ) )
{
unsigned char pbUncompressedData[PKTBUF_LEN-4];
int nUncompressedLength;
int nErrorCode;

// Initialize the max buffer size for zlib
nUncompressedLength = sizeof(pbUncompressedData);

// Inflate the packet
nErrorCode = InflatePacket( &data[4], (*len)-4, pbUncompressedData, &nUncompressedLength);

if( nErrorCode )
{
memmove( data+4, pbUncompressedData , nUncompressedLength );
*len = nUncompressedLength + 4;
}

return nErrorCode;
}

// Skip if its not an encrypted packet
if ( !( (opcode == CharProfileCode) || (opcode == ZoneSpawnsCode) ||
(opcode == NewSpawnCode)) )
return 0;

// Grab our player packet
if (queuePlayerProfile.size())
player = queuePlayerProfile[0];
else
player = NULL;

// Lock our mutex so the queues can't unwind before this packet is in
// NOTE: We only need to lock when the queue's are being used
if (locateActive)
pthread_mutex_lock(&mutexQueue);

prevKey = decodeKey;
// Try processing the packet first
#if HAVE_LIBEQ
if (player)
result = ProcessPacket(data, len, &decodeKey, cli, player->data, player->len);
else
result = ProcessPacket(data, len, &decodeKey, cli);
#else
result = 0;
#endif

if (result)
{
// Release our mutex
if (locateActive)
pthread_mutex_unlock(&mutexQueue);
// Check to see if we found a key
if ( prevKey != decodeKey)
FoundKey();
return(result);
}

// Queue it up... Allocate storage for our encrypted packet
if ((opcode == CharProfileCode) || (opcode == ZoneSpawnsCode))
pktrec = (EQPktRec *)malloc(sizeof(EQPktRec) + PKTBUF_LEN);
else if (opcode == NewSpawnCode)
pktrec = (EQPktRec *)malloc(sizeof(EQPktRec) + *len);

// Bail if we can't get memory...
if (!pktrec)
{
printf("Warning: Couldn't allocate memory to queue encrypted packet.\n");
// Release our mutex
if (locateActive)
pthread_mutex_unlock(&mutexQueue);
return(result);
}

// Grab the packet info
pktrec->len = *len;
memcpy(pktrec->data, data, *len);

// Queue the packet
switch (opcode)
{
case CharProfileCode:
queuePlayerProfile.insert(queuePlayerProfile.begin(), pktrec);
break;

case ZoneSpawnsCode:
queueZoneSpawns.insert(queueZoneSpawns.begin(), pktrec);
break;

case NewSpawnCode:
queueSpawns.insert(queueSpawns.begin(), pktrec);
break;
}

// Release the mutex and return if the locate thread is already running
if (locateActive)
{
pthread_mutex_unlock(&mutexQueue);
return(result);
}

// Start the locate thread if we have player/spawn packets available
if (queuePlayerProfile.size() && queueSpawns.size())
{
// Create the thread with standard scheduling parameters
pthread_create(&locateThread, NULL, ThreadFunc, this);

// And don't retain any results
pthread_detach(locateThread);

// Tell everyone we're active
locateActive = 1;
}

// Get outa here...

return(result);
}

int
EQDecode::InflatePacket (unsigned char *pbDataIn, int cbDataInLen, unsigned char* pbDataOut, int* pcbDataOutLen)
{
z_stream zstream;
int zerror = 0;
int i;

zstream.next_in = pbDataIn;
zstream.avail_in = cbDataInLen;
zstream.next_out = pbDataOut;
zstream.avail_out = *pcbDataOutLen;
zstream.zalloc = NULL;
zstream.zfree = NULL;
zstream.opaque = NULL;

i = inflateInit2( &zstream, 15 );
if (i != Z_OK)
{
*pcbDataOutLen = 0;
return 0;
}

zerror = inflate( &zstream, Z_FINISH );

if(zerror == Z_STREAM_END)
{
inflateEnd( &zstream );
*pcbDataOutLen = zstream.total_out;
return 1;
}
else
{
zerror = inflateEnd( &zstream );
*pcbDataOutLen = 0;
return 0;
}
}
